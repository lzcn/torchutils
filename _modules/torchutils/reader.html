[docs]class DataReader(metaclass=ABCMeta):
    """Callable reader for different types of data.

    Suppose data is saved in key-value manner.

    Args:
        path (str): data path
        transforms (Callable, optional): [description]. Defaults to lambdax:x

    Example::

        reader = DataReader(path)
        data = reader(key)

    """

    def __init__(self, path: str, transforms: lambda x: x):
        self.path = path
        self.transforms = transforms


[docs]    @abstractmethod
    def load(self, name):
        """Return data for given name."""
        pass


    def __call__(self, name):
        """Load data and transforms."""
        data = self.load(name)
        return self.transforms(data)

    @staticmethod
    def _open_lmdb(path):
        return lmdb.open(path, max_readers=1, readonly=True, lock=False, readahead=False, meminit=False)




[docs]class FeatLMDBReader(DataReader):
    """Reader for nn feature with LMDB backend."""

    def __init__(self, path, transforms=None):
        super().__init__(path, transforms=transforms)
        self._env = self._open_lmdb(path)


[docs]    def load(self, name):
        with self._env.begin(write=False) as txn:
            buf = txn.get(name.encode())
        feature = np.frombuffer(buf, dtype=np.float32).reshape(1, -1)
        return torch.from_numpy(feature).view(-1)

[docs]    def load(self, name):
        # read from raw image
        path = os.path.join(self.path, name)
        with open(path, "rb") as f:
            img = Image.open(f).convert("RGB")
        return img

[docs]    def load(self, name):
        with self._env.begin(write=False) as txn:
            imgbuf = txn.get(name.encode())
        buf = six.BytesIO()
        buf.write(imgbuf)
        buf.seek(0)
        img = Image.open(buf).convert("RGB")
        return img
